函数命名规则
 "S+系统名+功能名":"参数1;参数2..."          数据库发协议调用服务器函数/模块
 "D+系统名+功能名":"参数1;参数2..."          服务器发协议调用数据库函数/模块

           "id:uint32",

        ]
    },

    "method":{
        "STest1":"bool;int8;int16;uint32",

        "CTest2":"Friend;string;float;double;uint16",
        "CTest3":"int8[];uint8;bool;Friend[]",
        "STest4":"int8[];uint8;bool;Person[]"
    }
}
